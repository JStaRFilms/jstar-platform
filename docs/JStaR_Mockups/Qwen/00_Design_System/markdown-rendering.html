YouTube API Integration Example

// FR034: YouTube Trends Analysis
async function analyzeChannel(channelUrl) {
  const apiKey = process.env.YOUTUBE_API_KEY;
  const channelId = extractChannelId(channelUrl);
  
  // Get latest videos
  const videos = await fetch(
    `https://www.googleapis.com/youtube/v3/search?part=snippet&channelId=${channelId}&maxResults=50&order=date&key=${apiKey}`
  ).then(res => res.json());
  
  // Analyze performance metrics
  const insights = videos.items.map(video => {
    return {
      title: video.snippet.title,
      published: video.snippet.publishedAt,
      // Virality score calculation (FR035)
      viralityScore: calculateVirality({
        title: video.snippet.title,
        views: video.statistics.viewCount,
        engagement: video.statistics.likeCount / video.statistics.viewCount
      })
    };
  });
  
  return insights.sort((a, b) => b.viralityScore - a.viralityScore);
}
